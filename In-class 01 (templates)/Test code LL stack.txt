	cDoublyLinkedList<cPerson> myPL;
	cPerson P1;	P1.first = "Bob";
	cPerson P2;	P2.first = "Sally";
	cPerson P3; P3.first = "Robin";
	myPL.insertBeforeCurrent(P1);
	myPL.insertBeforeCurrent(P2);
	myPL.insertBeforeCurrent(P3);

	std::cout << "Bob, Sally, Robin" << std::endl;
	std::cout << "---------------" << std::endl;

	myPL.moveToFirst();
	for (int count = 0; count != 3; count++)
	{
		std::cout << myPL.getCurrent().first << std::endl;
		myPL.moveNext();
	}

	myPL.moveToFirst();
	myPL.deleteAtCurrent();
	std::cout << "---------------" << std::endl;
	myPL.moveToFirst();

	for (int count = 0; count != 2; count++)
	{
		std::cout << myPL.getCurrent().first << std::endl;
		myPL.moveNext();
	}
